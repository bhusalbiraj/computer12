SQL Operations

Logical Operations

  • AND: SELECT * FROM CUSTOMERS WHERE AGE > 30 AND COUNTRY='USA'
  • OR: SELECT * FROM ORDERS WHERE TOTAL_AMOUNT > 1000 OR STATUS='SHIPPED'
  • NOT: SELECT * FROM PRODUCTS WHERE NOT CATEGORY='ELECTRONICS'

Ordering Operations

  • ASC: SELECT * FROM EMPLOYEES ORDER BY HIRE_DATE ASC;
  • DESC: SELECT * FROM SALES ORDER BY AMOUNT DESC;

Grouping Operations

  • GROUP BY: SELECT CATEGORY, AVG(PRICE) FROM PRODUCTS GROUP BY CATEGORY;

Aggregate Functions

  • AVERAGE: SELECT AVG(SALARY) FROM EMPLOYEES;
  • MINIMUM: SELECT MIN(PRICE) FROM PRODUCTS;
  • MAXIMUM: SELECT MAX(AMOUNT) FROM SALES;
  • SUM: SELECT SUM(TOTAL_AMOUNT) FROM ORDERS;
  • COUNT: SELECT COUNT(*) FROM CUSTOMERS;
  • DISTINCT: SELECT DISTINCT(COUNTRY) FROM CUSTOMERS;
  • ROUND: SELECT ROUND(AVG(RATING), 2) FROM REVIEWS;

String Functions

  • LENGTH: SELECT LEN('BANANA');
  • REVERSE: SELECT REVERSE('HELLO');
  • REPLICATE: SELECT REPLICATE('ABC', 3);
  • UPPERCASE: SELECT UPPERCASE('hello world');
  • LOWERCASE: SELECT LOWERCASE('HELLO WORLD');
  • CONCATENATION: SELECT CONCAT('Hello, ', 'World!');
  • LEFT MOST CHARACTER: SELECT LEFT('BANANA', 3);
  • RIGHT MOST CHARACTER: SELECT RIGHT('BANANA', 3);

Other Operations

  • IN: SELECT * FROM CITIES WHERE COUNTRY IN ('USA', 'CANADA', 'MEXICO');
  • BETWEEN: SELECT * FROM ORDERS WHERE TOTAL_AMOUNT BETWEEN 500 AND 1000;
  • PERCENTAGE: SELECT * FROM PRODUCTS WHERE CATEGORY LIKE '%ELECTRONICS%';
  • UNDERSCORE: SELECT * FROM CUSTOMERS WHERE NAME LIKE '_A%';
  • HAVING: SELECT CATEGORY, AVG(PRICE) FROM PRODUCTS GROUP BY CATEGORY HAVING AVG(PRICE) > 100;
